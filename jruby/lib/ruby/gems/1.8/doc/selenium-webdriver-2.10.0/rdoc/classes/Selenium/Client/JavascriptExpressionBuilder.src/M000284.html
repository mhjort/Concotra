      def find_text_in_document(pattern, options)
        js_regexp = case pattern
        when Regexp
          pattern.inspect
        else
          /#{pattern}/.inspect
        end
        append "var text_match;\ntext_match = (null != selenium.browserbot.getCurrentWindow().document.body.innerHTML.match(\#{js_regexp}));\n"
        
      end
