        def stop
          return if @process.nil? || @process.exited?

          Net::HTTP.get uri.host, '/shutdown', uri.port
          @process.poll_for_exit STOP_TIMEOUT
        rescue ChildProcess::TimeoutError
          # ok, force quit
          @process.stop STOP_TIMEOUT
        end
