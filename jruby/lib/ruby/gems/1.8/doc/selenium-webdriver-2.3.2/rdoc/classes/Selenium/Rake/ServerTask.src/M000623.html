      def initialize(prefix = "selenium:server")
        @jar = nil
        @prefix = prefix
        @port = 4444
        @timeout = 30
        @background = true
        @log = true
        @opts = []

        yield self if block_given?

        unless @jar
          raise MissingJarFileError, "must provide path to the selenium server jar"
        end

        @server = Selenium::Server.new(@jar, :port       => @port,
                                             :timeout    => @timeout,
                                             :background => @background,
                                             :log        => @log )

        @server << @opts

        define_start_task
        define_stop_task
        define_restart_task
      end
