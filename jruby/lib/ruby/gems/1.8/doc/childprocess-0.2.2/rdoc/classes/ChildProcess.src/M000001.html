    def new(*args)
      case platform
      when :jruby
        JRuby::Process.new(args)
      when :ironruby
        IronRuby::Process.new(args)
      when :windows
        Windows::Process.new(args)
      when :macosx, :linux, :unix, :cygwin
        Unix::Process.new(args)
      else
        raise Error, "unsupported platform #{platform.inspect}"
      end
    end
