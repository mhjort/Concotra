    def obtain(klass, root = Object)
      Innate.sync do
        view_name = /^#{klass.to_s.downcase.dup.delete('_')}$/i
        if view = View.constants.grep(view_name).first
          root.const_get(view)
        else
          raise(NameError, "View #{klass} not found")
        end
      end
    end
