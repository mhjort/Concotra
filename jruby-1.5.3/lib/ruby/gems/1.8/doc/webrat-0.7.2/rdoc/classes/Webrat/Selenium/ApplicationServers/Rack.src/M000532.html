        def start
          @pid = fork do
            if File.exist?("log")
              redirect_io(STDOUT, File.join("log", "webrat.webrick.stdout.log"))
              redirect_io(STDERR, File.join("log", "webrat.webrick.stderr.log"))
            end
            exec start_command
          end
        end
