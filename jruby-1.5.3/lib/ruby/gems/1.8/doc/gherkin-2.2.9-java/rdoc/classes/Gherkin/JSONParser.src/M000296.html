    def step(o)
      multiline_arg = nil
      if(ma = o['multiline_arg'])
        if(ma['type'] == 'table')
          multiline_arg = rows(ma['value'])
        else
          multiline_arg = Formatter::Model::PyString.new(ma['value'], ma['line'])
        end
      end
      Formatter::Model::Step.new(comments(o), keyword(o), name(o), line(o), multiline_arg)
    end
