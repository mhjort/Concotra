        def to_hash
          hash = super
          if Array === @multiline_arg
            hash['multiline_arg'] = {
              'type' => 'table',
              'value' => hash['multiline_arg']
            }
          elsif PyString === @multiline_arg
            hash['multiline_arg']['type'] = 'py_string'
          end
          hash
        end
