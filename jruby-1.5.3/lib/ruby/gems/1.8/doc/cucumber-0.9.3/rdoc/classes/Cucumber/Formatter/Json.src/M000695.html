      def before_table_row(row)
        @current_row = {:cells => []}

        if @current_object.member? :examples
          @current_object[:examples][:table] << @current_row
        elsif @current_step
          (@current_step[:table] ||= []) << @current_row
        else
          internal_error
        end
      end
