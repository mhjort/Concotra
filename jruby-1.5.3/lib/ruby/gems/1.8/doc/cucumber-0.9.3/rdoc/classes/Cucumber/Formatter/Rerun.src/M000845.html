      def after_features(features)
        files = @file_names.uniq.map do |file|
          lines = @file_colon_lines[file]
          "#{file}:#{lines.join(':')}"
        end
        @io.puts files.join(' ')
        
        # Flusing output to rerun tempfile here...
        @io.flush
        @io.close
      end
