      def ensure_valid_keys(user_metadata)
        RESERVED_KEYS.each do |key|
          if user_metadata.keys.include?(key)
            raise "\#{\"*\"*50}\n:\#{key} is not allowed\n\nRSpec reserves some hash keys for its own internal use,\nincluding :\#{key}, which is used on:\n\n\#{caller(0)[4]}.\n\nHere are all of RSpec's reserved hash keys:\n\n\#{RESERVED_KEYS.join(\"\\n  \")}\n\#{\"*\"*50}\n"
            raise ":#{key} is not allowed"
          end
        end
      end
