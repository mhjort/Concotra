        def backtrace_line(line)
          return nil if configuration.cleaned_from_backtrace?(line)
          line = line.sub(File.expand_path("."), ".")
          line = line.sub(/\A([^:]+:\d+)$/, '\\1')
          return nil if line == '-e:1'
          line
        end
