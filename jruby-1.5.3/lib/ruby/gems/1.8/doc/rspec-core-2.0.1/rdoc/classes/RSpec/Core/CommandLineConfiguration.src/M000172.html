      def run
        case @command
        when 'autotest' then Autotest.generate
        else raise ArgumentError, "\n\#{\"*\"*50}\n\"\#{@command}\" is not valid a valid argument to \"rspec --configure\".\nSupported options are:\n\nrspec --configure autotest # generates configuration to run autotest with rspec\n\n\#{\"*\"*50}\n"
        end
      end
