      def unwrap_script_result(arg)
        if arg.kind_of?(Array)
          arg.map { |e| unwrap_script_result(e) }
        else
          if arg.kind_of?(Hash) && arg.member?("ELEMENT")
            Element.new self, element_id_from(arg)
          else
            arg
          end
        end
      end
