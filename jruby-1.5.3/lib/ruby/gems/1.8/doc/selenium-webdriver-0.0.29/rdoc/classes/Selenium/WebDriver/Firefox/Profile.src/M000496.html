        def self.from_json(json)
          zip_file = Tempfile.new("webdriver-profile-duplicate-#{json.hash}")

          zip_file << JSON.parse(json)['zip'].unpack("m")[0]
          zip_file.close

          new(Zipper.unzip(zip_file.path))
        ensure
          zip_file.delete if zip_file
        end
