        def create_element(&blk)
          element_ptr_ref = FFI::MemoryPointer.new :pointer
          yield element_ptr_ref

          auto_ptr = FFI::AutoPointer.new(element_ptr_ref.get_pointer(0), Lib.method(:finalize_element))
          Element.new(self, auto_ptr)
        end
