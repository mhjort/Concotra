      def send_keys(*args)
        args.each do |arg|
          case arg
          when Symbol
            arg = Keys[arg]
          when Array
            arg = arg.map { |e| e.kind_of?(Symbol) ? Keys[e] : e }.join
            arg << Keys[:null]
          end

          bridge.sendKeysToElement(@id, arg.to_s)
        end
      end
