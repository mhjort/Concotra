        def quit
          getWindowHandles.each do |handle|
            begin
              switchToWindow handle
              close
            rescue Error::NoSuchWindowError
            end
          end
        ensure
          ptr = @driver_pointer
          @driver_pointer = nil

          check_error_code Lib.wdFreeDriver(ptr), "unable to free driver #{self}"
        end
