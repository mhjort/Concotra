        def self.launcher(*args)
          launcher =  case Platform.os
                      when :windows
                        WindowsLauncher.new(*args)
                      when :macosx
                        MacOSXLauncher.new(*args)
                      when :unix, :linux
                        UnixLauncher.new(*args)
                      else
                        raise "unknown OS: #{Platform.os}"
                      end

          launcher
        end
