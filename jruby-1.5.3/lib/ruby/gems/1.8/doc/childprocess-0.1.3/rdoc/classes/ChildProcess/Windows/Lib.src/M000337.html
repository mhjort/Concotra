      def self.last_error_message
        errnum = get_last_error
        buf = FFI::MemoryPointer.new :char, 512

        size = format_message(
          FORMAT_MESSAGE_FROM_SYSTEM | FORMAT_MESSAGE_ARGUMENT_ARRAY,
          nil, errnum, 0, buf, buf.size, nil
        )

        buf.read_string(size).strip
      end
